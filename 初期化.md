### データベース作成
```
CREATE DATABASE sampledb;
```

### Memberテーブル
```
DROP TABLE IF EXISTS member;
CREATE TABLE member (
    id          INT UNSIGNED NOT NULL AUTO_INCREMENT,
    username   	VARCHAR(50),
    password   	VARCHAR(128),
    last_name  	VARCHAR(50),
    first_name 	VARCHAR(50),
    birthday   	CHAR(8),
    reg_date   	DATETIME,
    cancel      DATETIME,
    PRIMARY KEY (id)
);
```

### Prememberテーブル
```
DROP TABLE IF EXISTS premember;
CREATE TABLE premember (
        id          INT UNSIGNED NOT NULL AUTO_INCREMENT,
        username   	VARCHAR(50),
        password   	VARCHAR(128),
        last_name  	VARCHAR(50),
        first_name 	VARCHAR(50),
        birthday   	CHAR(8),
        link_pass   VARCHAR(128),
        reg_date   	DATETIME,
        PRIMARY KEY (id)
) ENGINE=INNODB;
```

### CharacterDataテーブル
```
DROP TABLE IF EXISTS characterdata;
CREATE TABLE characterdata (
    id          INT UNSIGNED NOT NULL AUTO_INCREMENT,
    name        VARCHAR(128),
    PRIMARY KEY (id)
)ENGINE=INNODB;
```

### Memoテーブル
```
DROP TABLE IF EXISTS memo;
CREATE TABLE memo (
    id              INT UNSIGNED NOT NULL AUTO_INCREMENT,
    member_id       INT UNSIGNED NOT NULL,
    character_id    INT UNSIGNED NOT NULL,
    flag            INT UNSIGNED NOT NULL,
    memo            VARCHAR(255),
    PRIMARY KEY (id),
    FOREIGN KEY(member_id) REFERENCES member(id) ON UPDATE CASCADE ON DELETE CASCADE,
    FOREIGN KEY(character_id) REFERENCES characterdata(id) ON UPDATE CASCADE ON DELETE CASCADE
)ENGINE=INNODB;
```
